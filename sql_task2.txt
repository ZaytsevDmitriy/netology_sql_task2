CREATE TABLE public.genre (
	id serial NOT NULL,
	genre_name varchar(30),
	CONSTRAINT genre_pkey PRIMARY KEY (id)
);


CREATE TABLE public.singer (
	id serial NOT NULL,
	"name" varchar(100),
	genre_id int,
	CONSTRAINT singer_name_key UNIQUE (name),
	CONSTRAINT singer_pkey PRIMARY KEY (id),
	CONSTRAINT genre_key FOREIGN KEY (genre_id) REFERENCES public.genre(id)
);

CREATE TABLE public.album (
	id serial NOT NULL,
	album_name varchar(50),
	year_of_issue int,
	id_singer int,
	CONSTRAINT album_pkey PRIMARY KEY (id),
	CONSTRAINT album_id_singer_fkey FOREIGN KEY (id_singer) REFERENCES public.singer(id)
);

CREATE TABLE public.track (
	id serial NOT NULL,
	track_name varchar(30),
	duration numeric,
	id_album int,
	CONSTRAINT track_pkey PRIMARY KEY (id),
	CONSTRAINT track_id_album_fkey FOREIGN KEY (id_album) REFERENCES public.album(id)
);
